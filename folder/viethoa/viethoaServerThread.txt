package t2;

import java.io.IOException;
import java.io.PrintWriter;
import java.net.ServerSocket;
import java.net.Socket;
import java.util.Scanner;

public class CapitalizeTool2ServerThread extends Thread {
	private static ServerSocket serverSocket;
	private static Socket socket;
	
	@Override
	public void run(){
		try {
			PrintWriter outServer = 
					new PrintWriter(socket.getOutputStream(),true);
			Scanner inServer = new Scanner(socket.getInputStream());
			String s = inServer.nextLine();
			String SS = s.toUpperCase();
			
			outServer.println(SS);
			socket.close();	
		} catch (IOException e) {
			// TODO Auto-generated catch block
			e.printStackTrace();
		}
		
	}
	
	public static void main(String[] args) throws Exception {
		serverSocket =new ServerSocket(2345);
		while(true) {
			socket = serverSocket.accept();
			CapitalizeTool2ServerThread ct = new CapitalizeTool2ServerThread();
			ct.start();
		}

	}
}
